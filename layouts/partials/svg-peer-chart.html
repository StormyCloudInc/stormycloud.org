{{- /* Generates SVG polyline points and area path for peer history charts. Input: dict "history" (array of {t, in, out}) "id" (string) "w" (int) "h" (int) "color" (string) */ -}} {{- $history := .history -}} {{- $id := .id -}} {{- $w := default 280 .w -}} {{- $h := default 80 .h -}} {{- $color := default "var(--accent)" .color -}} {{- $pad := 4.0 -}} {{- if and $history (gt (len $history) 0) -}} {{- $points := slice -}} {{- $max := 0.0 -}} {{- $min := 9999999.0 -}} {{- range $history -}} {{- $total := add (float .in) (float .out) -}} {{- if gt $total $max -}}{{- $max = $total -}}{{- end -}} {{- if lt $total $min -}}{{- $min = $total -}}{{- end -}} {{- $points = $points | append $total -}} {{- end -}} {{- $range := sub $max $min -}} {{- if eq $range 0.0 -}}{{- $range = 1.0 -}}{{- end -}} {{- $count := len $points -}} {{- $coords := slice -}} {{- range $i, $val := $points -}} {{- $x := 0.0 -}} {{- if eq $count 1 -}} {{- $x = div (float $w) 2.0 -}} {{- else -}} {{- $x = div (mul (float $i) (float $w)) (sub (float $count) 1.0) -}} {{- end -}} {{- $y := add $pad (div (mul (sub $max $val) (sub (float $h) (mul $pad 2.0))) $range) -}} {{- $coords = $coords | append (printf "%.1f,%.1f" $x $y) -}} {{- end -}} {{- /* Y-axis labels */ -}} {{- $fmtMax := "" -}} {{- $fmtMin := "" -}} {{- if ge $max 1000.0 -}} {{- $fmtMax = printf "%.0fk" (div $max 1000.0) -}} {{- else -}} {{- $fmtMax = printf "%.0f" $max -}} {{- end -}} {{- if ge $min 1000.0 -}} {{- $fmtMin = printf "%.0fk" (div $min 1000.0) -}} {{- else -}} {{- $fmtMin = printf "%.0f" $min -}} {{- end -}}
